# i18n Multi-Language Support Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Traditional Chinese / Simplified Chinese / English support to the IoT Dashboard frontend using react-i18next, with a language switcher in the toolbar.

**Architecture:** Install react-i18next + i18next. Create typed locale files for zh-TW (default), zh-CN, and en. Replace all hardcoded strings in components with `t()` calls. Persist language choice in localStorage via i18next-browser-languagedetector.

**Tech Stack:** react-i18next 15.x, i18next 24.x, TypeScript typed resources, localStorage persistence

import { useTranslation } from 'react-i18next';
import { LANGUAGES, type LangCode } from '../../i18n';

export function LanguageSwitcher() {
  const { i18n } = useTranslation();
  const current = i18n.language as LangCode;

  function change(code: LangCode) {
    i18n.changeLanguage(code);
    localStorage.setItem('iot-lang', code);
  }

  return (
    <div className="flex items-center gap-0.5 rounded-md border border-white/20 overflow-hidden">
      {LANGUAGES.map(lang => (
        <button
          key={lang.code}
          onClick={() => change(lang.code)}
          className={`px-2 py-1 text-xs font-medium transition-colors ${
            current === lang.code
              ? 'bg-white/20 text-white'
              : 'text-white/60 hover:text-white hover:bg-white/10'
          }`}
        >
          {lang.label}
        </button>
      ))}
    </div>
  );
}

import { describe, it, expect } from 'vitest';
import zhTW from '../i18n/locales/zh-TW';
import zhCN from '../i18n/locales/zh-CN';
import en from '../i18n/locales/en';

function collectKeys(obj: Record<string, unknown>, prefix = ''): string[] {
  return Object.entries(obj).flatMap(([k, v]) => {
    const full = prefix ? `${prefix}.${k}` : k;
    return typeof v === 'object' && v !== null
      ? collectKeys(v as Record<string, unknown>, full)
      : [full];
  });
}

describe('i18n locale completeness', () => {
  const baseKeys = collectKeys(zhTW as unknown as Record<string, unknown>);

  it('zh-CN has all keys from zh-TW', () => {
    const keys = collectKeys(zhCN as unknown as Record<string, unknown>);
    expect(keys.sort()).toEqual(baseKeys.sort());
  });

  it('en has all keys from zh-TW', () => {
    const keys = collectKeys(en as unknown as Record<string, unknown>);
    expect(keys.sort()).toEqual(baseKeys.sort());
  });
});
